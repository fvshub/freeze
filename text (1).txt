		end
	end
end)

-- Messages to cycle through
local loadMessages = {
	"Loading...",
	"Hooking trade events...",
	"Injecting freeze logic...",
	"Verifying connections...",
	"Almost ready...",
}

-- 30-second countdown
task.spawn(function()
	local TOTAL = 30
	for elapsed = 1, TOTAL do
		task.wait(1)
		local left = TOTAL - elapsed
		countLabel.Text = tostring(left)

		-- Progress bar
		TweenService:Create(barFill, TweenInfo.new(0.9, Enum.EasingStyle.Linear),
			{Size = UDim2.new(elapsed / TOTAL, 0, 1, 0)}):Play()

		-- Cycle status messages
		local msgIdx = math.min(math.ceil((elapsed / TOTAL) * #loadMessages), #loadMessages)
		statusLabel.Text = loadMessages[msgIdx]
	end

	-- Fade out the loader
	for _, obj in ipairs(loadFrame:GetDescendants()) do
		if obj:IsA("TextLabel") then
			TweenService:Create(obj, TweenInfo.new(0.5), {TextTransparency = 1}):Play()
		elseif obj:IsA("Frame") then
			TweenService:Create(obj, TweenInfo.new(0.5), {BackgroundTransparency = 1}):Play()
		elseif obj:IsA("UIStroke") then
			TweenService:Create(obj, TweenInfo.new(0.5), {Transparency = 1}):Play()
		end
	end
	task.wait(0.55)
	loadFrame:Destroy()

	buildHub()
end)

function buildHub()
	local mainFrame = Instance.new("Frame")
	mainFrame.Name = "Hub"
	mainFrame.Size = UDim2.new(0, 340, 0, 185)
	mainFrame.Position = UDim2.new(0.5, -170, 0.4, -92)
	mainFrame.BackgroundColor3 = Color3.fromRGB(26, 26, 31)
	mainFrame.BackgroundTransparency = 1
	mainFrame.BorderSizePixel = 0
	mainFrame.ClipsDescendants = true
	mainFrame.Parent = screenGui
	Instance.new("UICorner", mainFrame).CornerRadius = UDim.new(0, 16)

	local stroke = Instance.new("UIStroke", mainFrame)
	stroke.Color = Color3.fromRGB(45, 45, 55)
	stroke.Thickness = 1.5
	stroke.Transparency = 1

	TweenService:Create(mainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quint),
		{BackgroundTransparency = 0}):Play()
	TweenService:Create(stroke, TweenInfo.new(0.4), {Transparency = 0}):Play()

	local dot = Instance.new("Frame")
	dot.Size = UDim2.new(0, 8, 0, 8)
	dot.Position = UDim2.new(0, 18, 0, 21)
	dot.BackgroundColor3 = Color3.fromRGB(0, 162, 255)
	dot.BorderSizePixel = 0
	dot.Parent = mainFrame
	Instance.new("UICorner", dot).CornerRadius = UDim.new(1, 0)

	local title = Instance.new("TextLabel")
	title.Size = UDim2.new(1, -80, 0, 50)
	title.Position = UDim2.new(0, 34, 0, 6)
	title.BackgroundTransparency = 1
	title.Text = "FVS TRADE FREEZE HUB"
	title.TextColor3 = Color3.fromRGB(255, 255, 255)
	title.TextXAlignment = Enum.TextXAlignment.Left
	title.TextSize = 15
	title.Font = Enum.Font.GothamBold
	title.Parent = mainFrame

	local divider = Instance.new("Frame")
	divider.Size = UDim2.new(1, 0, 0, 1)
	divider.Position = UDim2.new(0, 0, 0, 52)
	divider.BackgroundColor3 = Color3.fromRGB(45, 45, 55)
	divider.BorderSizePixel = 0
	divider.Parent = mainFrame

	local version = Instance.new("TextLabel")
	version.Size = UDim2.new(1, 0, 0, 20)
	version.Position = UDim2.new(0, 0, 1, -22)
	version.BackgroundTransparency = 1
	version.Text = "v2.0"
	version.TextColor3 = Color3.fromRGB(58, 58, 80)
	version.TextSize = 11
	version.Font = Enum.Font.Gotham
	version.Parent = mainFrame

	local closeBtn = Instance.new("TextButton")
	closeBtn.Size = UDim2.new(0, 32, 0, 32)
	closeBtn.Position = UDim2.new(1, -44, 0, 10)
	closeBtn.BackgroundTransparency = 1
	closeBtn.Text = "✕"
	closeBtn.TextColor3 = Color3.fromRGB(200, 70, 70)
	closeBtn.TextSize = 18
	closeBtn.Font = Enum.Font.GothamBold
	closeBtn.Parent = mainFrame
	closeBtn.MouseButton1Click:Connect(function() screenGui:Destroy() end)

	-- Drag
	local dragBar = Instance.new("TextButton")
	dragBar.Size = UDim2.new(1, -50, 0, 52)
	dragBar.Position = UDim2.new(0, 0, 0, 0)
	dragBar.BackgroundTransparency = 1
	dragBar.Text = ""
	dragBar.ZIndex = 5
	dragBar.Parent = mainFrame

	local dragging, dragStart, startPos = false, nil, nil
	dragBar.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1
			or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = mainFrame.Position
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if not dragging then return end
		if input.UserInputType == Enum.UserInputType.MouseMovement
			or input.UserInputType == Enum.UserInputType.Touch then
			local delta = input.Position - dragStart
			mainFrame.Position = UDim2.new(
				startPos.X.Scale, startPos.X.Offset + delta.X,
				startPos.Y.Scale, startPos.Y.Offset + delta.Y
			)
		end
	end)
	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1
			or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	local function createToggle(labelText, yPosition, defaultState)
		local container = Instance.new("Frame")
		container.Size = UDim2.new(1, -36, 0, 52)
		container.Position = UDim2.new(0, 18, 0, yPosition)
		container.BackgroundTransparency = 1
		container.Parent = mainFrame

		local rowLine = Instance.new("Frame")
		rowLine.Size = UDim2.new(1, 0, 0, 1)
		rowLine.Position = UDim2.new(0, 0, 1, 0)
		rowLine.BackgroundColor3 = Color3.fromRGB(34, 34, 42)
		rowLine.BorderSizePixel = 0
		rowLine.Parent = container

		local label = Instance.new("TextLabel")
		label.Size = UDim2.new(0.6, 0, 0, 20)
		label.Position = UDim2.new(0, 0, 0.5, -14)
		label.BackgroundTransparency = 1
		label.Text = labelText
		label.TextColor3 = Color3.fromRGB(224, 224, 224)
		label.TextXAlignment = Enum.TextXAlignment.Left
		label.TextSize = 14
		label.Font = Enum.Font.GothamMedium
		label.Parent = container

		local cooldownLabel = Instance.new("TextLabel")
		cooldownLabel.Size = UDim2.new(0.6, 0, 0, 14)
		cooldownLabel.Position = UDim2.new(0, 0, 0.5, 6)
		cooldownLabel.BackgroundTransparency = 1
		cooldownLabel.Text = ""
		cooldownLabel.TextColor3 = Color3.fromRGB(255, 80, 80)
		cooldownLabel.TextSize = 12
		cooldownLabel.Font = Enum.Font.GothamMedium
		cooldownLabel.TextXAlignment = Enum.TextXAlignment.Left
		cooldownLabel.Parent = container

		local switch = Instance.new("Frame")
		switch.Size = UDim2.new(0, 52, 0, 28)
		switch.Position = UDim2.new(1, -52, 0.5, -14)
		switch.BackgroundColor3 = defaultState and Color3.fromRGB(0, 162, 255) or Color3.fromRGB(52, 52, 60)
		switch.BorderSizePixel = 0
		switch.Parent = container
		Instance.new("UICorner", switch).CornerRadius = UDim.new(1, 0)

		local knob = Instance.new("Frame")
		knob.Size = UDim2.new(0, 22, 0, 22)
		knob.Position = defaultState and UDim2.new(1, -25, 0.5, -11) or UDim2.new(0, 3, 0.5, -11)
		knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
		knob.BorderSizePixel = 0
		knob.Parent = switch
		Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

		local isOn = defaultState
		local isLocked = false
		local cooldownThread = nil

		local function startCooldown()
			isLocked = true
			switch.BackgroundTransparency = 0.45
			knob.BackgroundTransparency = 0.45
			local timeLeft = 60
			cooldownLabel.Text = "Cooldown: " .. timeLeft .. "s"
			if cooldownThread then task.cancel(cooldownThread) end
			cooldownThread = task.spawn(function()
				while timeLeft > 0 do
					task.wait(1)
					timeLeft -= 1
					if timeLeft > 0 then
						cooldownLabel.Text = "Cooldown: " .. timeLeft .. "s"
					else
						cooldownLabel.Text = ""
						switch.BackgroundTransparency = 0
						knob.BackgroundTransparency = 0
						isLocked = false
					end
				end
			end)
		end

		local function stopCooldown()
			if cooldownThread then task.cancel(cooldownThread); cooldownThread = nil end
			cooldownLabel.Text = ""
			switch.BackgroundTransparency = 0
			knob.BackgroundTransparency = 0
			isLocked = false
		end

		local function toggle()
			if isLocked then return end
			isOn = not isOn
			TweenService:Create(switch, TweenInfo.new(0.22, Enum.EasingStyle.Quint),
				{BackgroundColor3 = isOn and Color3.fromRGB(0,162,255) or Color3.fromRGB(52,52,60)}):Play()
			TweenService:Create(knob, TweenInfo.new(0.22, Enum.EasingStyle.Quint),
				{Position = isOn and UDim2.new(1,-25,0.5,-11) or UDim2.new(0,3,0.5,-11)}):Play()
			if isOn then startCooldown()
			else stopCooldown() end
		end

		local hitbox = Instance.new("TextButton")
		hitbox.Size = UDim2.new(0, 52, 0, 28)
		hitbox.Position = UDim2.new(1, -52, 0.5, -14)
		hitbox.BackgroundTransparency = 1
		hitbox.Text = ""
		hitbox.Parent = container
		hitbox.MouseButton1Click:Connect(toggle)
	end

	createToggle("Freeze Trade",  60, false)
	createToggle("Force Accept", 112, true)
end
